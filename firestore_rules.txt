
// Firestore Rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper functions
    function isSignedIn() { return request.auth != null; }
    function getUserRole() { return get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role; }
    function getUserClientId() { return get(/databases/$(database)/documents/users/$(request.auth.uid)).data.clientId; }
    function isManufacturer() { return getUserRole() in ['support_agent', 'supervisor', 'admin']; }

    match /users/{userId} {
      allow read: if isSignedIn();
      allow write: if isManufacturer() || request.auth.uid == userId;
    }

    match /tickets/{ticketId} {
      allow read: if isSignedIn() && (isManufacturer() || resource.data.clientId == getUserClientId());
      allow create: if isSignedIn();
      allow update: if isSignedIn() && (
        isManufacturer() || (resource.data.clientId == getUserClientId() && request.resource.data.status == 'Closed')
      );
    }

    match /comments/{commentId} {
      allow read: if isSignedIn() && (isManufacturer() || get(/databases/$(database)/documents/tickets/$(resource.data.ticketId)).data.clientId == getUserClientId());
      allow create: if isSignedIn();
    }
  }
}

// Storage Rules
service firebase.storage {
  match /b/{bucket}/o {
    match /attachments/{ticketId}/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.resource.size < 10 * 1024 * 1024;
    }
  }
}
